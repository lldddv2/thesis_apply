\tcbuselibrary{theorems, skins, breakable}

% Contador compartido entre entornos (numeración por sección)
\newcounter{formalcounter}[section]
\renewcommand{\theformalcounter}{\thesection.\arabic{formalcounter}}

% Registrar tipos de label para cleveref (requerido por label type=...)
\crefname{definicion}{definición}{definiciones}
\Crefname{definicion}{Definición}{Definiciones}
\crefname{teorema}{teorema}{teoremas}
\Crefname{teorema}{Teorema}{Teoremas}
\crefname{observacion}{observación}{observaciones}
\Crefname{observacion}{Observación}{Observaciones}

% -- Definición --------------------------------------------------
\newtcbtheorem[
    use counter=formalcounter
]{definicion}{Definición}{
    enhanced,
    breakable,
    colback=blue!4!white,
    colframe=blue!50!black,
    fonttitle=\bfseries,
    separator sign={ —},
    attach boxed title to top left={yshift=-2mm, xshift=4mm},
    boxed title style={
        colback=blue!50!black,
        colframe=blue!50!black,
    },
    label type=definicion
}{def}

% ============================================================
% FIGURAS
% Uso: \myFigure[fig:label]{filename}[caption][width]
%   fig:label  (opcional) — label para \ref{}
%   filename   (obligatorio) — nombre del archivo en figures/
%   caption    (opcional) — pie de figura
%   width      (opcional, default 0.5) — fracción de \textwidth
% ============================================================
\NewDocumentCommand{\myFigure}{O{} m O{} O{0.5}}{
    \begin{figure}[H]
        \centering
        \includegraphics[width=#4\textwidth]{#2}
        \caption{#3}
        \label{#1}
    \end{figure}
}
\numberwithin{figure}{section}


% ============================================================
% TABLAS DESDE CSV
% Uso: \myTable[tab:label]{csvfile}[caption]
%   tab:label  (opcional) — label para \ref{}
%   csvfile    (obligatorio) — path al archivo CSV
%   caption    (opcional) — pie de tabla
% ============================================================
\makeatletter
\csvset{
  autotabularcenter/.style={
    file=#1,
    after head=\csv@pretable\begin{tabular}{*{\csv@columncount}{c}}\csv@tablehead,
    table head=\toprule\csvlinetotablerow\\\midrule,
    late after line=\\,
    table foot=\\\bottomrule,
    late after last line=\csv@tablefoot\end{tabular}\csv@posttable,
    command=\csvlinetotablerow},
}
\makeatother

\NewDocumentCommand{\myTable}{O{} m O{}}{
    \begin{table}[H]
        \centering
        \csvloop{autotabularcenter={#2}}
        \caption{#3}
        \label{#1}
    \end{table}
}
\numberwithin{table}{section}
